[docs]def continuous(age, Z, imf, iso, sl, neb, fracNeb=0., inLsun=False,
               with_units=False, nthreads=__NTHREADS__):
    """
    Generate an SSP assuming a continuously sampled mass function

    Parameters
    ----------

    age: float
        age of the population (in yr)

    Z: float
        metallicity

    imf: imf.Imf instance
        Initial Mass function

    iso: isochrone.Isochrone instance
        Isochrone library

    sl: stellib.Stellib instance
        stellar library

    neb: nebular.Nebular instance
        Nebular Model

    fracNeb: float
        Photon fraction tranformed into nebular component

    inLsun: bool
        return results units normalized to Lsun

    with_units: bool
        if set, return results with units

    nthreads:  int
        number of threads during interpolation

    Returns
    -------
    l: numpy.ndarray[float, ndim=1]
        wavelength in AA

    s: numpy.ndarray[float, ndim=1]
        Spectrum in ergs/s/AA or ergs/s/AA/Lsun

    p: dict
        various intermediate results and population properties
    """
    # check units
    _age = val_in_unit('age', age, 'yr')

    # generate a sample of stars
    stars  = iso._get_continuous_isochrone(_age, Z)
    # masses = numpy.power(10., stars['logM'])
    Mtot   = 1.  # normalized by def  integral (masses dm ) = 1
    Lbol   = (10 ** stars['logL']).sum()

    # compute their contributions according to the Mass Function
    # dN = imf(M) * dM
    # dN    = imf.nstars1(stars['logM'], dlogm=stars['dlogm'])
    # dN    = imf.nstars1(stars['logM'])
    dN    = imf.nstars(stars['logM'])

    # generate stellar spectrum
    #  * interpolate the spectral library to build the spectrum of each star
    #  * sum the contributions according to the IMF
    r, l0, s0, NHI = genStellarSpectrum(stars, Z, sl, weights=dN,
                                        inLsun=inLsun, with_units=False,
                                        nthreads=nthreads)

    # add Nebular component
    if neb is not None:
        if inLsun:
            l, s = neb.add_emission(l0 * unit['AA'], s0 * Lsun, NHI, fracNeb)
            s /= Lsun
        else:
            l, s = neb.add_emission(l0 * unit['AA'], s0, NHI, fracNeb)
    else:
        l = l0
        s = s0

    # concatenate the info
    properties = dict(Nstars=0,
                      inLsun=inLsun,
                      age=age,
                      Z=Z,
                      fracneb=fracNeb,
                      imf=imf.name,
                      iso=iso.name,
                      stellib=sl.name,
                      stars=stars,
                      Mtot=Mtot,
                      Lbol=Lbol,
                      r=r,
                      l0=l0,
                      s0=s0
                      )

    if with_units:
        l = l * unit['angstrom']
        s = s * unit['erg/s/angstrom/Msun']
        properties['s0'] = s0 * unit['erg/s/angstrom/Msun']
        if inLsun:
            s = s / unit['lsun']
            properties['s0'] = properties['s0'] / unit['lsun']

        properties['age'] = _age
        properties['l0'] = l0 * unit['angstrom']
        properties['Mtot'] = Mtot * unit['msun']
        properties['Lbol'] = Lbol * unit['lsun']

        return l, s, properties
    else:
        return l, s, properties

                  nsamp=20, inLsun=False, with_units=True,
                  nthreads=__NTHREADS__):
    """
    Generate an SSP assuming a continuously sampled mass function and
    boostrap the mass binning according to the number of stars.
    The number of samples is set by nsamp

    The approximation we use in this modeling scheme is that stars are binned
    in mass to their continuous equivalent mass bins and represented by their
    central star spectrum.  This approximation is good enough as long as
    variations from on side of the mass bin to the other are negligeable.

    Parameters
    ----------
    nstars: int
        total number of stars in the population

    age: float
        age of the population (in yr)

    Z: float
        metallicity

    imf: imf.Imf instance
        Initial Mass function

    iso: isochrone.Isochrone instance
        Isochrone library

    sl: stellib.Stellib instance
        stellar library

    neb: nebular.Nebular instance
        Nebular Model

    fracNeb: float
        Photon fraction tranformed into nebular component

    dlogm: float
        log mass step used to sample the isochrone

    dlogt: float
        log temperature used to sample the isochrone

    dlogl: float
        log luminosity used to sample the isochrone

    nsamp: int
        number of mass sampling

    inLsun: bool
        return results units normalized to Lsun

    with_units: bool
        if set, return results with units

    nthreads:  int
        number of threads during interpolation

    Returns
    -------
    l: numpy.ndarray[float, ndim=1]
        wavelength in AA

    s: numpy.ndarray[float, ndim=1]
        Spectrum in ergs/s/AA or ergs/s/AA/Lsun

    p: dict
        various intermediate results and population properties



    .. todo::
        Could be optimized to compute only once the interpolation on the grid
        and apply variable weights ==> lot more sampling for the same time
    """
    # check units
    _age = val_in_unit('age', age, 'yr')

    # Continuously sample the corresponding isochrone
    # -----------------------------------------------
    # get reference stars for a continuously sampled isochrone
    stars, logm_bins = iso._get_continuous_isochrone(_age, Z, ret_bins=True, dlogm=dlogm)
    m_bins = 10 ** logm_bins

    if (np.diff(m_bins) < 0).any():
        raise Exception('binning error')

    # filter the masses according to the IMF definition
    # -------------------------------------------------
    # imf_truncation = (stars['logM'] >= log10(imf.massMin)) & (stars['logM'] <= log10(imf.massMax))
    # logM = stars['logM'][imf_truncation]
    # dlogM = stars['dlogm'][imf_truncation]
    # stars = stars[imf_truncation]
    logM = stars['logM']
    # dlogM = stars['dlogm']

    # find the corresponding bin edges
    # --------------------------------
    # bins1 = 10 ** ( np.hstack([ logM[0] - 0.5 * dlogM[0], logM + 0.5 * dlogM ]))

    # print 'bins', len(m_bins)
    # print 'dbin', m_bins - bins1
    # print 'dbin', np.abs(m_bins - bins1).max()

    # Simulate the continuous population
    # ----------------------------------
    # This will set all the dimensions for the variables

    # dN, bins = imf.nstars1(stars['logM'], dlogm=stars['dlogm'], ret_bins=True)
    dN = imf.nstars(stars['logM'], ret_bins=False, edges=logm_bins)

    # minM = 10 ** stars['logM'].min()
    maxM = 10 ** stars['logM'].max()

    r, l0, s0, NHI = genStellarSpectrum(stars, Z, sl, weights=dN, inLsun=inLsun, nthreads=nthreads)
    if neb is not None:
        if inLsun:
            l, s = neb.add_emission(l0 * unit['angstrom'], s0 * Lsun, NHI, fracNeb)
            s /= Lsun
        else:
            l, s = neb.add_emission(l0 * unit['angstrom'], s0, NHI, fracNeb)
    else:
        l = l0
        s = s0

    # for book keeping
    ll0 = l0[:]
    ls0 = s0[:]

    # Simulate nsamp discrete populations
    # -----------------------------------

    # book keeping variables
    ds   = np.zeros((nsamp, s.size), dtype=float)
    ds0  = np.zeros((nsamp, s0.size), dtype=float)
    dn   = np.zeros((nsamp, len(dN)), dtype=float)
    Mtot = np.zeros(nsamp, dtype=float)
    Minit = np.zeros(nsamp, dtype=float)
    Mdead = np.zeros(nsamp, dtype=float)
    Mmax = np.zeros(nsamp, dtype=float)
    Lbol = np.zeros(nsamp, dtype=float)

    # reset stellib weights
    r[:, 1] *= 0

    for k in Pbar('Sampling').iterover(range(nsamp)):
        # Mass properties of the populations
        # ----------------------------------
        # _masses = imf.random(nstars, minM, maxM)
        _masses = imf.random(nstars)
        ind = (_masses > maxM)
        Mdead[k] = np.sum(_masses[ind])   # total mass above the isochrone limit
        Minit[k] = np.sum(_masses)
        Mmax[k] = np.max(_masses)
        # _masses = _masses[~ind]
        Mtot[k] = Minit[k]

        # Approximation scheme
        # --------------------
        # n, edges = histogram( val, bins=b)
        _dN, _ = np.histogram( _masses, bins=m_bins)

        # Generate the stellar spectrum
        # -----------------------------
        # note: no need to recompute the weights through the interpolation
        _r, l0, s0, NHI = genStellarSpectrum(stars, Z, sl,
                                             weights=(_dN).astype(float),
                                             inLsun=inLsun, nthreads=nthreads)
        r[:, 1] += _r[:, 1]

        # add Nebular component
        # ---------------------
        if neb is not None:
            if inLsun:
                l, s = neb.add_emission(l0 * unit['angstrom'], s0 * Lsun, NHI, fracNeb)
                s /= Lsun
            else:
                l, s = neb.add_emission(l0 * unit['angstrom'], s0, NHI, fracNeb)
        else:
            l = l0
            s = s0

        # book keeping
        dn[k]  = _dN[:]
        ds[k]  = s
        ds0[k] = s0
        Lbol[k] = (10 ** stars['logL'] * _dN.astype(float)).sum()

    # s      = np.median(ds, axis=0)
    # s0     = np.median(ds0, axis=0)
    s      = np.average(ds, axis=0)
    s0     = np.average(ds0, axis=0)
    del ds0

    # concatenate the info
    properties = dict(dn=dn,
                      m=10 ** logM,
                      m_bins=m_bins,
                      dm=np.diff(m_bins),
                      ds=ds,
                      Nstars=nstars,
                      inLsun=inLsun,
                      age=age,
                      Z=Z,
                      fracneb=fracNeb,
                      imf=imf.name,
                      iso=iso.name,
                      stellib=sl.name,
                      stars=stars,
                      Mtot=Mtot,
                      Lbol=Lbol,
                      r=r,
                      l0=l0,
                      s0=s0,
                      Mdead=Mdead,
                      Minit=Minit,
                      Mmax=Mmax,
                      dnc=dN,
                      ll0=ll0,
                      ls0=ls0
                      )

    if with_units:
        l = l * unit['angstrom']
        s = s * unit['erg/s/angstrom']
        properties['s0'] = s0 * unit['erg/s/angstrom']
        properties['ds'] = ds * unit['erg/s/angstrom']
        properties['ls0'] = ls0 * unit['erg/s/angstrom']
        if inLsun:
            s = s / unit['lsun']
            properties['s0'] = properties['s0'] / unit['lsun']
            properties['ds'] = properties['ds'] / unit['lsun']
            properties['ls0'] = properties['ls0'] / unit['lsun']

        properties['age'] = _age
        properties['l0'] = l0 * unit['angstrom']
        properties['ll0'] = ll0 * unit['angstrom']
        properties['Mtot'] = Mtot * unit['msun']
        properties['Mdead'] = Mdead * unit['msun']
        properties['Minit'] = Minit * unit['msun']
        properties['Mmax'] = Mmax * unit['msun']
        properties['m'] = 10 ** logM * unit['msun']
        properties['m_bins'] = m_bins * unit['msun']
        properties['dm'] = np.diff(m_bins) * unit['msun']
        properties['Lbol'] = Lbol * unit['lsun']

        return l, s, properties
    else:
        return l, s, properties




[docs]def discrete(nstars, age, Z, imf, iso, sl, neb, fracNeb=0., nsamp=20,
             inLsun=False, with_units=True, nthreads=__NTHREADS__):
    """
    Generate a stochastic SSP assuming a discretely sampled mass function
    according to the number of stars.
    The number of samples is set by nsamp

    Parameters
    ----------
    nstars: int
        total number of stars in the population

    age: float
        age of the population (in yr)

    Z: float
        metallicity

    imf:   imf.Imf
        Initial Mass function

    iso: isochrone.Isochrone
        Isochrone library

    sl: stellib.Stellib
        stellar library

    neb: nebular.Nebular
        Nebular Model

    fracNeb: float
        Photon fraction tranformed into nebular component

    nsamp: int
        number of mass sampling

    inLsun: bool
        return results units normalized to Lsun

    with_units: bool
        if set, return results with units

    nthreads: int
        number of threads during interpolation

    Returns
    -------
    l: numpy.ndarray[float, ndim=1]
        wavelength (in AA, by default)

    s: numpy.ndarray[float, ndim=1]
        Spectrum in ergs/s/AA or ergs/s/AA/Lsun

    p: dict
        various intermediate results and population properties

    .. todo::
        Could be optimized to compute only once the interpolation on the
        grid and apply variable weights ==> lot more sampling for the same time
    """
    # check units
    u_age = val_in_unit('age', age, 'yr')

    Mtot = np.empty(nsamp, dtype=float)
    masses = np.empty((nsamp, nstars), dtype=float)
    r = {}

    for k in range(nsamp):
        _masses = imf.random(nstars)
        pop     = iso._get_isochrone(u_age, Z, masses=_masses)
        Mtot[k] = _masses.sum()
        _r, l0, s0, NHI = genStellarSpectrum(pop, Z, sl, inLsun=inLsun, nthreads=nthreads)
        for rk in _r:
            if rk[0] in r:
                r[rk[0]] += rk[1]
            else:
                r[rk[0]] = rk[1]
        # add Nebular component
        if neb is not None:
            if inLsun:
                l, s = neb.add_emission(l0 * unit['angstrom'], s0 * Lsun, NHI, fracNeb)
                s /= Lsun
            else:
                l, s = neb.add_emission(l0 * unit['angstrom'], s0, NHI, fracNeb)
        else:
            l = l0
            s = s0

        if k == 0:
            ds   = np.zeros((nsamp, s.size))
            ds0  = np.zeros((nsamp, s0.size))

        ds[k]  = s
        ds0[k] = s0
        masses[k] = _masses

    Lbol   = (np.power(10., pop['logL'])).sum()
    s      = np.average(ds, axis=0)
    s0     = np.average(ds0, axis=0)
    r      = np.array(list(r.items()))
    del ds0

    # concatenate the info
    properties = dict(dn=1.,
                      dm=None,  # numpy.diff(bins),
                      ds=ds,
                      Nstars=nstars,
                      inLsun=inLsun,
                      age=age,
                      Z=Z,
                      fracneb=fracNeb,
                      imf=imf.name,
                      iso=iso.name,
                      stellib=sl.name,
                      stars=pop,
                      Mtot=Mtot,
                      Lbol=Lbol,
                      r=r,
                      l0=l0,
                      s0=s0,
                      masses=masses
                      )

    if with_units:
        l = l * unit['angstrom']
        s = s * unit['erg/s/angstrom']
        properties['s0'] = s0 * unit['erg/s/angstrom']
        properties['ds'] = ds * unit['erg/s/angstrom']
        if inLsun:
            s = s / unit['lsun']
            properties['s0'] = properties['s0'] / unit['lsun']
            properties['ds'] = properties['ds'] / unit['lsun']

        properties['age'] = u_age
        properties['l0'] = l0 * unit['angstrom']
        properties['Mtot'] = Mtot * unit['msun']
        properties['Lbol'] = Lbol * unit['lsun']
        properties['masses'] = masses * unit['msun']

        return l, s, properties
    else:
        return l, s, properties




[docs]def fast_discrete1(nstars, age, Z, imf, iso, sl, neb, fracNeb=0., dlogm=0.001,
                   nsamp=20, inLsun=False, with_units=True, full_outputs=False,
                   nthreads=__NTHREADS__):
    """
    Generate an SSP assuming a continuously sampled mass function and
    boostrap the mass binning according to the number of stars.
    The number of samples is set by nsamp

    The approximation we use in this modeling scheme is that stars are
    binned in mass to their continuous equivalent mass bins and represented
    by their central star spectrum.  This approximation is good enough as
    long as variations from on side of the mass bin to the other are
    negligeable.

    This function is optimized to make large collections of clusters at fixed
    isochrone.

    Parameters
    ----------
    nstars: int
        total number of stars in the population

    age: float
        age of the population (in yr)

    Z: float
        metallicity

    imf: imf.Imf instance
        Initial Mass function

    iso: isochrone.Isochrone instance
        Isochrone library

    sl: stellib.Stellib instance
        stellar library

    neb: nebular.Nebular instance
        Nebular Model

    fracNeb: float
        Photon fraction tranformed into nebular component

    nsamp: int
        number of mass sampling

    inLsun: bool
        return results units normalized to Lsun

    with_units: bool
        if set, return results with units

    full_outputs: bool
        if set, returns also the individual stars and their frequencies

    Returns
    -------
    l: numpy.ndarray[float, ndim=1]
        wavelength in AA

    s: numpy.ndarray[float, ndim=1]
        Spectrum in ergs/s/AA or ergs/s/AA/Lsun

    p: dict
        various population properties

    stars: Table
        table of stars describing the isochrone
        only if full_output is set

    specs: ndarray
        spectra associated to each star in stars
        only if full_output is set

    dN: ndarray
        number of stars represented by each of the ones in stars
        only if full_output is set

    """
    # check units
    _age = val_in_unit('age', age, 'yr').magnitude

    Lsun = float(unit['lsun'].to('erg/s').magnitude)

    # Continuously sample the corresponding isochrone
    # and simulate the continuous population
    # to set the different variables and dimensions, respectively
    # -----------------------------------------------------------
    # get reference stars for a continuously sampled isochrone
    stars, logm_bins = iso._get_continuous_isochrone(_age, Z, ret_bins=True,
                                                     dlogm=dlogm)
    m_bins = 10 ** logm_bins
    # try to add robustness
    if (np.diff(m_bins) < 0).any():
        raise Exception('binning error')

    # get the predicted number of stars per bin of mass on this isochrone
    # dN = imf.nstars(stars['logM'], ret_bins=False, edges=logm_bins)
    maxM = 10 ** stars['logM'].max()

    # compute all the individual spectra incl. neb component.
    # this will allow us to to a simple weighted sum.
    stars_lamb, stars_specs, NHI = genIndividualStellarSpectrum(stars, Z, sl,
                                                                inLsun=inLsun)

    # add the nebular emission
    if (neb is not None) & (fracNeb > 0):
        if inLsun:
            l, s = neb.add_emission_to_many(stars_lamb * unit['angstrom'],
                                            stars_specs * Lsun, NHI, fracNeb)
            s /= Lsun
        else:
            l, s = neb.add_emission_to_many(stars_lamb * unit['angstrom'],
                                            stars_specs, NHI, fracNeb)
    else:
        l = stars_lamb
        s = stars_specs

    # book keeping variables
    specs = np.zeros((nsamp, l.size), dtype=float)
    Mtot = np.zeros(nsamp, dtype=float)
    Minit = np.zeros(nsamp, dtype=float)
    Mdead = np.zeros(nsamp, dtype=float)
    Mmax = np.zeros(nsamp, dtype=float)
    Lbol = np.zeros(nsamp, dtype=float)
    age = np.ones(nsamp, dtype=float) * _age
    Z = np.ones(nsamp, dtype=float) * Z
    if full_outputs:
        dN = np.zeros((nsamp, len(stars)), dtype=float)

    for k in Pbar('Mixing').iterover(range(nsamp)):
        # Mass properties of the populations
        # ----------------------------------
        # _masses = imf.random(nstars, minM, maxM)
        _masses = imf.random(nstars)

        # Approximation scheme
        # --------------------
        _dN = histogram( _masses, m_bins)[0].astype(float)

        specs[k, :] = (_dN[:, None] * s).sum(axis=0)
        Lbol[k] = (10 ** stars['logL'] * _dN).sum()
        ind = (_masses > maxM)
        # total mass above the isochrone limit
        Mdead[k] = np.sum(_masses[ind])
        Minit[k] = np.sum(_masses)
        Mmax[k] = np.max(_masses)
        _masses = _masses[~ind]
        Mtot[k] = Minit[k]
        if full_outputs:
            dN[k, :] = _dN

    if with_units:
        l = l * unit['angstrom']
        specs = specs * unit['erg/s/angstrom']
        if inLsun:
            specs = specs / unit['lsun']

        properties = dict(Mtot=Mtot * unit['msun'],
                          Minit=Minit * unit['msun'],
                          Mdead=Mdead * unit['msun'],
                          Mmax=Mmax * unit['msun'],
                          Lbol=Lbol * unit['lsun'],
                          age=age * unit['yr'],
                          Z=Z)
    else:
        properties = dict(Mtot=Mtot, Minit=Minit, Mdead=Mdead, Mmax=Mmax,
                          Lbol=Lbol, age=age, Z=Z)

    if full_outputs:
        return l, specs, properties, stars, s, dN
    else:
        return l, specs, properties
